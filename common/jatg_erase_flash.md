# jatg_erase_flash
```bash
#Jtag Cmd
#RomWriteEnbale
smn 0x5A3A4 0x6
smn 0x5A39C 0x0
# wait 0x5A3A0=0x1
sleep 1
smn 0x5A3A0
sleep 1
smn 0x5A3A0 0x0

# Eease 0x0 addr
smn 0x5A3A4 0x20
smn 0x5A39C 0x30000
# wait 0x5A3A0=0x1
sleep 1
smn 0x5A3A0
sleep 1
smn 0x5A3A0 0x0

sleep 1
#WaitRomWriteBusy
smn 0x5A3A4 0x5
smn 0x5A39C 0x40001

# wait 0x5A3A0=0x1
sleep 1
smn 0x5A3A0
sleep 1
smn 0x5A3A0 0x0

# wait 0x5A3A8=0x1
smn 0x5A3A8
```
